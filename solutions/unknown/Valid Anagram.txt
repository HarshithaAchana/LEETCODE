// Title: Valid Anagram
            // Difficulty: Unknown
            // Language: Python3

class Solution:
    def isAnagram(self, s: str, t: str) -> bool:
        a_Array=list(s)
        b_Array=list(t)
        if len(a_Array) != len(b_Array) :
            return False
        for i in a_Array :
            if   i in b_Array:
                b_Array.remove(i)
        if len(b_Array) == 0 : 
            return True
        else :
            return False
